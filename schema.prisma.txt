// ============================================================
// KASIR SAAS — PostgreSQL Schema via Prisma ORM
// General Ledger Architecture: Every financial event writes
// a debit/credit entry. Balance = SUM(credits) - SUM(debits)
// ============================================================

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ─── USERS (TENANT) ────────────────────────────────────────
model User {
  id           String   @id @default(cuid())
  telegramId   String   @unique @map("telegram_id")
  name         String
  username     String?
  avatarUrl    String?  @map("avatar_url")
  pinHash      String   @map("pin_hash")          // bcrypt hash of 6-digit PIN
  isActive     Boolean  @default(true) @map("is_active")
  memberSince  DateTime @default(now()) @map("member_since")
  expiredAt    DateTime @map("expired_at")
  notifiedH3   Boolean  @default(false) @map("notified_h3") // H-3 notification flag
  createdAt    DateTime @default(now()) @map("created_at")
  updatedAt    DateTime @updatedAt @map("updated_at")

  // Relations
  ledgerEntries  LedgerEntry[]
  products       Product[]
  categories     TransactionCategory[]
  unitTypes      UnitType[]

  @@map("users")
}

// ─── GENERAL LEDGER (CORE ACCOUNTING) ──────────────────────
// Every financial event is an immutable row here.
// Balance = SUM(amount WHERE type=CREDIT) - SUM(amount WHERE type=DEBIT)
model LedgerEntry {
  id          String          @id @default(cuid())
  userId      String          @map("user_id")
  type        LedgerType      // CREDIT (masuk) | DEBIT (keluar)
  amount      Decimal         @db.Decimal(15, 2)
  balanceAfter Decimal        @db.Decimal(15, 2) @map("balance_after") // snapshot saldo setelah transaksi
  categoryId  String          @map("category_id")
  description String?
  refType     RefType?        @map("ref_type")  // POS_SALE | STOCK_PURCHASE | MANUAL
  refId       String?         @map("ref_id")    // FK ke sale/stock record jika ada
  recordedAt  DateTime        @default(now()) @map("recorded_at") // timestamp presisi
  createdAt   DateTime        @default(now()) @map("created_at")

  // Relations
  user        User                  @relation(fields: [userId], references: [id], onDelete: Cascade)
  category    TransactionCategory   @relation(fields: [categoryId], references: [id])
  saleItems   SaleItem[]

  @@index([userId, recordedAt])
  @@index([userId, type])
  @@map("ledger_entries")
}

enum LedgerType {
  CREDIT // Uang masuk (pendapatan, tambah modal)
  DEBIT  // Uang keluar (pengeluaran, beli stok, prive)
}

enum RefType {
  POS_SALE        // Dari transaksi POS
  STOCK_PURCHASE  // Bayar supplier / tambah stok
  MANUAL_INCOME   // Tambah modal / pemasukan manual
  OPERATIONAL     // Biaya operasional
  WITHDRAWAL      // Tarik tunai / prive
}

// ─── TRANSACTION CATEGORIES (DYNAMIC) ──────────────────────
model TransactionCategory {
  id          String      @id @default(cuid())
  userId      String      @map("user_id")
  name        String      // e.g. "Penjualan Produk", "Biaya Iklan"
  type        LedgerType  // CREDIT atau DEBIT
  isDefault   Boolean     @default(false) @map("is_default")
  createdAt   DateTime    @default(now()) @map("created_at")

  user          User          @relation(fields: [userId], references: [id], onDelete: Cascade)
  ledgerEntries LedgerEntry[]

  @@unique([userId, name])
  @@map("transaction_categories")
}

// ─── PRODUCTS (INVENTORY) ──────────────────────────────────
model Product {
  id            String    @id @default(cuid())
  userId        String    @map("user_id")
  name          String
  description   String?
  imageUrl      String?   @map("image_url")
  categoryName  String?   @map("category_name")  // free-text product category
  unitTypeId    String    @map("unit_type_id")
  buyPrice      Decimal   @db.Decimal(15, 2) @map("buy_price")    // Harga Modal
  sellPrice     Decimal   @db.Decimal(15, 2) @map("sell_price")   // Harga Jual
  stock         Int       @default(0)
  isActive      Boolean   @default(true) @map("is_active")
  createdAt     DateTime  @default(now()) @map("created_at")
  updatedAt     DateTime  @updatedAt @map("updated_at")

  user          User          @relation(fields: [userId], references: [id], onDelete: Cascade)
  unitType      UnitType      @relation(fields: [unitTypeId], references: [id])
  stockLogs     StockLog[]
  saleItems     SaleItem[]

  @@index([userId])
  @@map("products")
}

// ─── UNIT TYPES (CUSTOM PER USER) ──────────────────────────
model UnitType {
  id        String    @id @default(cuid())
  userId    String    @map("user_id")
  name      String    // Pcs, Dus, Pak, Lusin, dll
  createdAt DateTime  @default(now()) @map("created_at")

  user      User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  products  Product[]

  @@unique([userId, name])
  @@map("unit_types")
}

// ─── STOCK LOGS (AUDIT TRAIL STOK) ─────────────────────────
model StockLog {
  id          String      @id @default(cuid())
  productId   String      @map("product_id")
  changeQty   Int         @map("change_qty")    // + tambah, - kurang
  stockAfter  Int         @map("stock_after")   // snapshot stok setelah
  reason      StockReason
  ledgerRefId String?     @map("ledger_ref_id") // FK ke LedgerEntry jika ada pengeluaran
  note        String?
  recordedAt  DateTime    @default(now()) @map("recorded_at")

  product     Product     @relation(fields: [productId], references: [id], onDelete: Cascade)

  @@index([productId, recordedAt])
  @@map("stock_logs")
}

enum StockReason {
  PURCHASE    // Beli dari supplier
  SALE        // Terjual via POS
  ADJUSTMENT  // Koreksi manual
  RETURN      // Retur barang
}

// ─── SALE ITEMS (POS TRANSACTIONS) ─────────────────────────
// Satu transaksi POS bisa multi-item. 
// Header transaksi = LedgerEntry dengan refType=POS_SALE
model SaleItem {
  id            String    @id @default(cuid())
  ledgerEntryId String    @map("ledger_entry_id")
  productId     String    @map("product_id")
  qty           Int
  priceAtSale   Decimal   @db.Decimal(15, 2) @map("price_at_sale")   // Harga jual saat itu
  buyPriceAtSale Decimal  @db.Decimal(15, 2) @map("buy_price_at_sale") // Modal saat itu (untuk hitung profit)
  subtotal      Decimal   @db.Decimal(15, 2)
  createdAt     DateTime  @default(now()) @map("created_at")

  ledgerEntry   LedgerEntry @relation(fields: [ledgerEntryId], references: [id], onDelete: Cascade)
  product       Product     @relation(fields: [productId], references: [id])

  @@index([ledgerEntryId])
  @@map("sale_items")
}
